// ✅ Reusable fetch wrapper — src/utils/safeFetch.ts
export async function safeFetch<T>(url: string, timeoutMs = 20_000): Promise<T> {
  let res: Response;

  try {
    res = await fetch(url, { signal: AbortSignal.timeout(timeoutMs) });
  } catch (e: any) {
    // AbortError або network failure
    throw new Error('Немає зв'язку з сервером. Перевір інтернет.');
  }

  if (!res.ok) {
    const body = await res.json().catch(() => null);
    const msg  = body?.error ?? `Помилка сервера: ${res.status}`;
    throw new Error(msg); // тепер це "Kodik timeout" а не "<html>502</html>"
  }

  return res.json() as Promise<T>;
}
